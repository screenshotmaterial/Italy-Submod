				}
			}
		}

		fire_only_once = yes

		ai_will_do = {
			factor = 1000
		}

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision LOAS_Invite_Country target: [From.GetName]"
			if = {
				limit = { tag = SYR }
				FROM = { country_event = syrfor.1 }
			}
			if = {
				limit = { tag = IRQ }
				FROM = { country_event = irqfor.4 }
			}
		}
	}
