MON_remove_russian_diplo_relations = {
	if = {
		limit = { has_idea = MON_russian_eco_aid }
		remove_ideas = MON_russian_eco_aid
	}
	if = {
		limit = { has_non_aggression_pact_with = RUS }
		diplomatic_relation = {
			country = RUS
			relation = non_aggression_pact
			active = no
		}
	}
	if = {
		limit = { has_military_access_to = RUS }
		diplomatic_relation = {
			country = RUS
			relation = military_access
			active = no
		}
	}
	RUS = {
		if = {
			limit = { has_guaranteed = ROOT }
			diplomatic_relation = {
				country = ROOT
				relation = guarantee
				active = no
			}
		}
		if = {
			limit = { has_military_access_to = ROOT }
			diplomatic_relation = {
				country = ROOT
				relation = military_access
				active = no
			}
		}
	}
}

MON_puppet_setup = {
	if = {
		limit = {
			is_russian_tag = yes
			has_socialist_government = no
			MON = { has_character = MON_roman_von_ungernsternberg }
		}
		MON = {
			promote_character = MON_roman_von_ungernsternberg
			set_politics = {
				ruling_party = paternal_autocrat
				elections_allowed = no
			}
			set_political_party = {
				ideology = paternal_autocrat
				popularity = 60
			}
			set_cosmetic_tag = MON_sternberg
			set_party_name = {
				ideology = paternal_autocrat
				name = MON_paternal_autocrat_party
				long_name = MON_paternal_autocrat_party_long
			}
			load_focus_tree = {
				tree = Mongolia_Sternberg
				keep_completed = yes
			}
		}
	}
	else_if = {
		limit = { is_chinese_tag = yes }
		#Setup MAN
		if = {
			limit = {
				is_ruled_by_kmt = yes
				NOT = { tag = XSM }
			}
			MON = {
				remove_ideas = {
					MON_russian_eco_aid
					MON_russian_military_mission
				}
				if = {
					limit = {
						ROOT = { is_ruled_by_right_kmt = yes }
					}
					MON = { set_cosmetic_tag = MON_RKMT }
				}
				else = { set_cosmetic_tag = MON_KMT }
				set_politics = {
					ruling_party = social_democrat
					elections_allowed = no
				}
				set_political_party = {
					ideology = social_democrat
					popularity = 60
				}
				load_focus_tree = {
					tree = Mongolia_Puppet
					keep_completed = yes
				}
			}
		}
		else = {
			MON = {
				remove_ideas = {
					MON_russian_eco_aid
					MON_russian_military_mission
				}
				set_party_name = {
					ideology = paternal_autocrat
					long_name = XSM_syndicalist_party_long
					name = XSM_syndicalist_party
				}
				set_politics = {
					ruling_party = paternal_autocrat
					elections_allowed = no
				}
				set_political_party = {
					ideology = paternal_autocrat
					popularity = 60
				}
				promote_character = MON_jodbajab
				load_focus_tree = {
					tree = Mongolia_Puppet
					keep_completed = yes
				}
			}
		}
	}
	else = {
		MON = {
			remove_ideas = {
				MON_russian_eco_aid
				MON_russian_military_mission
			}
			promote_character = MON_jodbajab
			set_party_name = {
				ideology = paternal_autocrat
				name = XSM_syndicalist_party
				long_name = XSM_syndicalist_party_long
			}
			load_focus_tree = {
				tree = Mongolia_Puppet
				keep_completed = yes
			}
		}
	}
}
