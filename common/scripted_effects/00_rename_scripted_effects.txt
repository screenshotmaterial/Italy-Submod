enable_renaming = {
	set_global_flag = allowrenaming_flag

	### set endo states ###
	every_state = {
		meta_effect = {
			text = { set_state_name = [STATENAME] }
			STATENAME = "endo_state_[THIS.GetId]"
		}
		for_each_loop = {
			array = state_victory_points
			meta_effect = {
				text = {
					set_province_name = { id = [VPID] name = [VPNAME] }
				}
				VPID = "[?v]"
				VPNAME = "endo_vp_[?v]"
			}
		}
	}

	do_all_renames = yes
}

disable_renaming = {
	clr_global_flag = allowrenaming_flag

	### reset endo states ###
	every_state = {
		reset_state_name = yes
		for_each_loop = {
			array = state_victory_points
			reset_province_name = var:v
		}
	}
}
