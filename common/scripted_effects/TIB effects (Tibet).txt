TIB_puppet_setup = {
	#Setup Ma Biao
	if = {
		limit = { tag = XSM }
		TIB = {
			set_party_name = {
				ideology = paternal_autocrat
				long_name = XSM_syndicalist_party_long
				name = XSM_syndicalist_party
			}
			set_politics = {
				ruling_party = paternal_autocrat
				elections_allowed = no
			}
			add_popularity = {
				ideology = paternal_autocrat
				popularity = 0.5
			}
			event_target:XSM_Ma_Biao_target = {
				set_nationality = PREV
			}
			if = {
				limit = { has_character = XSM_Ma_Biao }
				XSM_Ma_Biao = {
					if = {
						limit = { has_ideology = paternal_autocrat_subtype }
						promote_character = { ideology = paternal_autocrat_subtype }
					}
					else = {
						add_country_leader_role = {
							country_leader = {
								desc = XSM_ma_biao_commander_desc
								ideology = paternal_autocrat_subtype
							}
							promote_leader = yes
						}
					}
				}
			}
		}
	}
	#Setup Abdullah Bughra
	else_if = {
		limit = { tag = ETS }
		TIB = {
			set_party_name = {
				ideology = national_populist

					if = {
						limit = { has_ideology = national_populist_subtype }
						promote_character = { ideology = national_populist_subtype }
					}
					else = {
						add_country_leader_role = {
							country_leader = {
								ideology = national_populist_subtype
								traits = { cornered_fox }
							}
							promote_leader = yes
						}
					}
				}
			}
		}
	}
	#Setup He Guoguang
	else_if = {
		limit = { is_ruled_by_pro_zhili = yes }
		TIB = {
			set_party_name = {
				ideology = authoritarian_democrat

				popularity = 0.5
			}
			event_target:SZC_He_Guoguang = {
				set_nationality = TIB
				promote_leader = yes
			}
			if = {
				limit = { has_character = SZC_he_guoguang }
				SZC_he_guoguang = {
					if = {
						limit = { has_ideology = authoritarian_democrat_subtype }
						promote_character = { ideology = authoritarian_democrat_subtype }
					}
					else = {
						add_country_leader_role = {
							country_leader = {
								desc = SZC_he_guoguang_leader_desc
								ideology = authoritarian_democrat_subtype
							}
							promote_leader = yes
						}
					}
				}
			}
		}
	}
	#Setup Wen Tsungyao
	else_if = {
		limit = { is_ruled_by_pro_manchu = yes }
		TIB = {
			set_party_name = {
				ideology = authoritarian_democrat
				long_name = XSM_syndicalist_party_long
				name = XSM_syndicalist_party

							country_leader = { ideology = authoritarian_democrat_subtype }
							promote_leader = yes
						}
					}
				}
			}
		}
	}
	#Setup Federalists
	else_if = {
		limit = { is_ruled_by_federalists = yes }
		TIB = {
			set_politics = {
				ruling_party = social_democrat
				elections_allowed = no
			}
			add_popularity = {
				ideology = social_democrat
				popularity = 0.5
			}
		}
	}
	#Setup Tibetan KMT with Huang Musong
	else_if = {
		limit = { is_ruled_by_kmt = yes }
		TIB = {
			if = {
				limit = {
					ROOT = { is_ruled_by_right_kmt = yes }
				}
				TIB = { set_cosmetic_tag = TIB_RKMT }
			}
			else = { set_cosmetic_tag = TIB_KMT }
			set_party_name = {
				ideology = paternal_autocrat

					}

						add_country_leader_role = {
							country_leader = {
								ideology = paternal_autocrat_subtype
								desc = TIB_huang_musong_country_leader_desc
							}
							promote_leader = yes
						}
					}
				}
			}
		}
	}
}
