UKR_remove_king = {
	hidden_effect = {
		if = {
			limit = { has_character = UKR_wilhelm_franz_von_habsburg }
			event_target:UKR_wilhelm_franz_von_habsburg_target = {
				set_character_name = "Wilhelm Franz von Habsburg-Lothringen"
				set_nationality = AUS
			}
		}
		if = {
			limit = { has_character = UKR_leo_stefan_von_habsburg }
			event_target:UKR_leo_stefan_von_habsburg_target = {
				set_character_name = "Leo Stefan von Habsburg-Lothringen"
				set_nationality = AUS
			}
		}
		if = {
			limit = { has_character = UKR_danylo_skoropadskyi }
			retire_character = UKR_danylo_skoropadskyi
		}
	}
}

UKR_create_vasyl = {
	if = {
		limit = { has_event_target = UKR_wilhelm_franz_von_habsburg_target }
		hidden_effect = {
			event_target:UKR_wilhelm_franz_von_habsburg_target = { set_nationality = PREV }
		}
		UKR_wilhelm_franz_von_habsburg = {
			set_character_name = "Vasyl"
			if = {
				limit = { has_ideology = social_democrat_subtype }
				promote_character = { ideology = social_democrat_subtype }
			}
			else = {
				add_country_leader_role = {
					country_leader = {
						desc = POLITICS_VASYL_DESC
						ideology = social_democrat_subtype
					}
					promote_leader = yes
				}
			}
			if = {
				limit = { has_ideology = social_liberal_subtype }
				promote_character = { ideology = social_liberal_subtype }
			}
			else = {
				add_country_leader_role = {
					country_leader = {
						desc = POLITICS_VASYL_DESC
						ideology = social_liberal_subtype
					}
					promote_leader = yes
				}
			}
			if = {
				limit = { has_ideology = market_liberal_subtype }
				promote_character = { ideology = market_liberal_subtype }
			}
			else = {
				add_country_leader_role = {
					country_leader = {
						desc = POLITICS_VASYL_DESC
						ideology = market_liberal_subtype
					}
					promote_leader = yes
				}
			}
			if = {
				limit = { has_ideology = social_conservative_subtype }
				promote_character = { ideology = social_conservative_subtype }
			}
			else = {
				add_country_leader_role = {
					country_leader = {
						desc = POLITICS_VASYL_DESC
						ideology = social_conservative_subtype
					}
					promote_leader = yes
				}
			}
			if = {
				limit = { has_ideology = authoritarian_democrat_subtype }

			}
			else = {

					country_leader = {

					}
					promote_leader = yes
				}
			}
			if = {
				limit = { has_ideology = paternal_autocrat_subtype}
				promote_character = { ideology = paternal_autocrat_subtype }
			}
			else = {
				add_country_leader_role = {
					country_leader = {
						desc = POLITICS_VASYL_DESC
						ideology = paternal_autocrat_subtype
					}
					promote_leader = yes

	else = {
		UKR_create_leo_stefan = yes
	}
