POR_remove_monarchy = {
	hidden_effect = {
		if = {
			limit = { has_cosmetic_tag = POR_monarchy }
			drop_cosmetic_tag = yes
		}
		if = {
			limit = { has_character = POR_duarte_braganca }
			POR_duarte_braganca = {
				set_character_name = "Duarte Nuno de Bragança"
				retire = yes
			}
		}
		if = {
			limit = { has_character = POR_maria_adelaide_braganca }
			POR_maria_adelaide_braganca = {
				set_character_name = "Maria Adelaide de Bragança"
				retire = yes
			}
		}
		if = {
			limit = { has_character = POR_maria_ana_braganca }
			POR_maria_ana_braganca = {
				set_character_name = "Maria Ana de Bragança"
				retire = yes
			}
		}
	}
}

POR_create_duarte = {
	if = {
		limit = { has_event_target = POR_duarte_braganca_target }
		if = {
			limit = { NOT = { has_character = POR_duarte_braganca } }
			event_target:POR_duarte_braganca_target = { set_nationality = PREV }
		}
		POR_duarte_braganca = {
			set_character_name = "Duarte II"
			if = {
				limit = { has_ideology = social_democrat_subtype }
				promote_character = { ideology = social_democrat_subtype }
			}
			else = {
				add_country_leader_role = {
					country_leader = {
						desc = POLITICS_DUARTE_BRAGANCA_DESC
						ideology = social_democrat_subtype
					}
					promote_leader = yes
				}
			}
			if = {
				limit = { has_ideology = social_liberal_subtype }
				promote_character = { ideology = social_liberal_subtype }
			}
			else = {
				add_country_leader_role = {
					country_leader = {
						desc = POLITICS_DUARTE_BRAGANCA_DESC
						ideology = social_liberal_subtype
					}
					promote_leader = yes
				}
			}
			if = {
				limit = { has_ideology = market_liberal_subtype }
				promote_character = { ideology = market_liberal_subtype }
			}
			else = {
				add_country_leader_role = {
					country_leader = {
						desc = POLITICS_DUARTE_BRAGANCA_DESC
						ideology = market_liberal_subtype
					}
					promote_leader = yes
				}
			}
			if = {
				limit = { has_ideology = social_conservative_subtype }
				promote_character = { ideology = social_conservative_subtype }
			}
			else = {
				add_country_leader_role = {
					country_leader = {
						desc = POLITICS_DUARTE_BRAGANCA_DESC
						ideology = social_conservative_subtype
					}
					promote_leader = yes
				}
			}
			if = {
				limit = { has_ideology = authoritarian_democrat_subtype }
				promote_character = { ideology = authoritarian_democrat_subtype }
			}
			else = {
				add_country_leader_role = {
					country_leader = {
						desc = POLITICS_DUARTE_BRAGANCA_DESC
						ideology = authoritarian_democrat_subtype
					}
					promote_leader = yes
				}
			}
			if = {
				limit = { has_ideology = paternal_autocrat_subtype }
				promote_character = { ideology = paternal_autocrat_subtype }
			}
			else = {
				add_country_leader_role = {
					country_leader = {
						desc = POLITICS_DUARTE_BRAGANCA_DESC
						ideology = paternal_autocrat_subtype
					}
					promote_leader = yes
				}
			}
			if = {
				limit = { has_ideology = national_populist_subtype }
				promote_character = { ideology = national_populist_subtype }
			}
			else = {
				add_country_leader_role = {
					country_leader = {
						desc = POLITICS_DUARTE_BRAGANCA_DESC
						ideology = national_populist_subtype
					}
					promote_leader = yes
				}
			}
		}
	}
	else = {
		POR_create_maria_adelaide = yes
	}
}
