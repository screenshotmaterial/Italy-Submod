CEN_conflict_escalates = {
	hidden_effect = {
		if = {
			limit = {
				NOT = {
					has_event_target = CEN_socialist_leader
					has_event_target = CEN_liberal_leader
					has_event_target = CEN_military_leader
				}
			}
			every_country = {
				limit = {
					OR = {
						tag = ELS
						tag = GUA
						tag = HON
						tag = NIC
					}
				}
				activate_mission = CEN_conflict_escalates_mission
			}
		}
	}
}
