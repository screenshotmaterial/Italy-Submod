on_actions = {
	on_ruling_party_change = {
		effect = {
			if = {
				limit = { original_tag = BUK }
				clear_sic_slot = yes
				if = {
					limit = { will_have_socialist_government = yes }
					activate_advisor = BUK_yuldash_akhunbabaev
				}
				else_if = {
					limit = { will_have_authoritarian_government = yes }
					activate_advisor = BUK_ibrahim_bek
				}
				else = {
					activate_advisor = BUK_mahmudkhodja_behbudiy
				}
			}
			else_if = {
				limit = { original_tag = KHI }
				clear_sic_slot = yes
				if = {
					limit = { will_have_socialist_government = yes }
					activate_advisor = KHI_kaikhaziz_atabayev
				}
				else_if = {
					limit = { will_have_authoritarian_government = yes }
					activate_advisor = KHI_eshi_khan
				}
				else = {
					activate_advisor = KHI_bekjon_rahmon
				}
			}
			else_if = {
				limit = { original_tag = TRK }
				clear_sic_slot = yes
				if = {
					limit = { will_have_socialist_government = yes }
					activate_advisor = TRK_arif_alimov
				}
				else_if = {
					limit = { will_have_authoritarian_government = yes }
					activate_advisor = TRK_katta_ergash
				}
				else = {
					activate_advisor = TRK_munawwar_qari_abdurrashidkhan
				}
			}
		}
	}
}
