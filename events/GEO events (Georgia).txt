		hidden_effect = { complete_national_focus = GEO_election_1936 }
	}

	option = { # Mensheviks win
		name = geo.120.a
		ai_chance = {
			base = 1
			modifier = {
				factor = 0

				factor = 0
				has_game_rule = {
					rule = GEO_election
					option = GEO_ELECTION_MONARCHY
				}

				factor = 0
				has_game_rule = {
					rule = GEO_election
					option = GEO_ELECTION_NATIONALIST
				}
			}
		}
		if = {
			limit = {
				NOT = {
					has_country_leader = { character = GEO_ioachim_von_hohenzollern }
					has_country_leader = { character = GEO_giorgi_bagration_mukhraneli }
				}
			}
			complete_national_focus = GEO_SSDP_victory
		}
		clr_country_flag = GEO_do_1936_election_flag
		add_popularity = {
			ideology = social_democrat
			popularity = 0.1
		}
		meta_effect = {
			text = {
				set_politics = {
					ruling_party = social_democrat
					last_election = "[CURRENT_DATE]"
					election_frequency = 48
					elections_allowed = yes
				}
			}
			CURRENT_DATE = "[?global.date.GetDateStringNoHour]"
		}
		country_event = { # coalition?
			id = geo.121
		}
	}

	option = { # EDP-MMEP coalition
		name = geo.120.b
		ai_chance = {
			base = 1
			modifier = {
				factor = 0
				OR = {
					has_game_rule = {
						rule = GEO_election
						option = GEO_ELECTION_MENSHEVIK_RADSOC
					}
					has_game_rule = {
						rule = GEO_election
						option = GEO_ELECTION_MENSHEVIK_SOCLIB
					}
				}
			}
		}
		if = {
			limit = {
				NOT = {
					has_country_leader = { character = GEO_ioachim_von_hohenzollern }
					has_country_leader = { character = GEO_giorgi_bagration_mukhraneli }
				}
			}
			complete_national_focus = GEO_EDP_MMEP_coalition
		}
		clr_country_flag = GEO_do_1936_election_flag
		add_popularity = {
			ideology = social_conservative
			popularity = 0.1
		}
		add_popularity = {
			ideology = market_liberal
			popularity = 0.05
		}
		meta_effect = {
			text = {
				set_politics = {

					election_frequency = 48
