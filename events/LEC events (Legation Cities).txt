country_event = { # Intro
	id = lec.1
	immediate = { log = "[GetDateText]: [Root.GetName]: event lec.1 FROM = [From.GetName]" }
	title = lec.1.t
	desc = lec.1.d
	picture = GFX_report_event_LEC_shanghai_river

	is_triggered_only = yes

	option = {
		name = lec.1.a
		effect_tooltip = {
			every_other_country = {
				limit = { has_country_flag = legation_council_vote }
				country_event = { id = lec.2 days = 3 }
			}
			every_other_country = {
				limit = {
					has_country_flag = legation_council_member
					NOT = { has_country_flag = legation_council_vote }
				}
				country_event = { id = lec.3 days = 3 }
			}
		}
		hidden_effect = {
			for_each_scope_loop = {
				array = global.LEC_council_members
				if = {
					limit = { has_country_flag = legation_council_vote }
					country_event = { id = lec.2 days = 3 }
				}
				else = {
					country_event = { id = lec.3 days = 3 }
				}
			}
		}
	}
}
